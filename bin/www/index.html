Overview

JSONP allows you to make cross-domain requests as long as the server you are requesting from will respond with JSON

Example

Do JSONP

The following was loaded using JSONP:

    The Code

    <p><a href="#" id="tryJSONP" class="greenButton">Do JSONP</a></p>
<p class="hidden">The following was loaded using JSONP:</p>
<ul id="jsonpTarget" class="plastic hidden"></p>
<script type="text/javascript">
	// http://develop.github.com/p/repo.html
	GitHubAPI = {};

	GitHubAPI.Repos = function(username, callback){
		requestURL = "http://github.com/api/v2/json/repos/show/" + username + "?callback=?";
		jQuery.getJSON(requestURL, function(json, status){
			callback(json.repositories, status);
		});
	};

	jQuery("#tryJSONP").bind("click", function(e){
		e.preventDefault();
		jQuery('#jsonp :hidden').show();
		jQuery('#jsonpTarget').empty().html('<li>loading...</li>');

		GitHubAPI.Repos("cmcculloh", function(json, status){
			var content = [];
			jQuery.each(json, function(i){
				content.push('<li class="project">');
				content.push(this['name']);
				content.push('</li>');
			});
			jQuery("#jsonpTarget").html(content.join(''));
		});
	});
</script>

Overview

Yahoo Pipes can digest any URL and spit out contents you specify (to a certain extent) as JSON

Example

Do Yahoo Pipe

The following was loaded using JSONP from Yahoo pipes:

    The Code

    <p><a href="#" id="tryYahooPipes" class="greenButton">Do Yahoo Pipe</a></p>
<ul id="yahooPipesTarget" class="rounded hidden"></ul>

<script type="text/javascript">
	yahooPipes = {};

	yahooPipes.getDeals = function(callback){
		requestURL = "http://pipes.yahoo.com/pipes/pipe.run?_id=5c2b38b0d1c8cc3a2439662af944d212&_render=json&_callback=?";
		jQuery.getJSON(requestURL, function(json, status){
			console.dir(json);
			callback(json.value.items, status);
		});
	};

	jQuery("#tryYahooPipes").bind("click", function(e){
		e.preventDefault();
		jQuery('#yahooPipes :hidden').show();
		jQuery('#yahooPipesTarget').empty().html('<li>loading...</li>');

		yahooPipes.getDeals(function(json, status){
			var theHTML, $titles;

			theHTML = [];
			contentHTML = json[0].content;

			//sanitize. Remove fancy quotes and make lower case.
			contentHTML = jQuery.replaceByCharCode(contentHTML);
			contentHTML = contentHTML.toLowerCase();
			contentHTML = '<div>' + contentHTML + '</div>';

			$deals = jQuery(contentHTML).find('div.dd2-sub-wrap');

			$deals.each(function(i, deal){
				var title, price, link;

				$deal = jQuery(deal);
				title = $deal.find("span.title").html();
				price = $deal.find("div.dd2-todaysprice").html().replace("today's price: ", "");
				link = $deal.find("a").attr('href');

				theHTML.push('<li class="arrow">'
							, '<a href="' , link , '">'
							, title
							, '</a>'
							, '<small class="counter">'
							, price
							, '</small>'
							, '</li>');
			});
			jQuery("#yahooPipesTarget").html(theHTML.join(''));
		});
	});

	;(function($){
		jQuery.replaceByCharCode=function(n){
			var split_n = n.split('');
			for(var i=0, ii=split_n.length;i<ii;i++){
				var charCode = split_n[i].charCodeAt(0);
				if(jQuery.replaceByCharCode.replaceBy[charCode]){
					console.log('found one!');
					split_n[i] = jQuery.replaceByCharCode.replaceBy[charCode];
				}
			}

			return split_n.join('');
		};

		jQuery.replaceByCharCode.replaceBy = {
			8217: "'",
			146: "'",
			151: "'"
		};
	})(jQuery);
</script>
